 # Static Variable Code # 

 what is Static Variable =
 - Static variable means one for a class.
 - Static variable get change the memory with class loader by class loaded.
 - Static variable get change the value with the particular object of the class.
 - Static function are recommended access to static variable.
 - Static variable are intilize using static block.
 - Static block get id has execution.
 - Static function are non-overhidden function.

 # Static Variable Code = 


   package practice;

class BankAc 
{
	int acNo;
	String holderName;
	double balance;
	static double intrestRate;
	
	static {
		intrestRate = 3.2;
	}
	
	public BankAc() {
		// TODO Auto-generated constructor stub
	}

	public BankAc(int acNo, String holderName, double balance) {
		super();
		this.acNo = acNo;
		this.holderName = holderName;
		this.balance = balance;
	}

	public int getAcNo() {
		return acNo;
	}

	public void setAcNo(int acNo) {
		this.acNo = acNo;
	}

	public String getHolderName() {
		return holderName;
	}

	public void setHolderName(String holderName) {
		this.holderName = holderName;
	}

	public double getBalance() {
		return balance;
	}

	public void setBalance(double balance) {
		this.balance = balance;
	}

	public static double getIntrestRate() {
		return intrestRate;
	}

	public static void setIntrestRate(double intrestRate) {
		BankAc.intrestRate = intrestRate;
	}

	@Override
	public String toString() {
		return "AcNo=" + acNo + ", HolderName=" + holderName + ", Balance=" + balance + ", Intrest Rate"+intrestRate;
	}
	
	
}//class BankAc ends here

public class Demo {
	
	public static void main(String[] args) {
		
		BankAc b1 = new BankAc();
		b1.acNo = 1234;
		b1.holderName = "Ritik";
		b1.balance = 25000;
		System.out.println(b1);
		
		BankAc b2 = new BankAc(5975,"Radhe",50000);
		System.out.println(b2);
		
		BankAc b3 = new BankAc(1404,"Shyam",30000);
		System.out.println(b3);
		
		b3.setIntrestRate(2.6);
		System.out.println(b3);
		
	}
}
